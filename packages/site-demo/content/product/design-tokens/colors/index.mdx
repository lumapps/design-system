import DesignTokenGroup from '@lumx/demo/layout/DesignTokenGroup';
import DesignToken from '@lumx/demo/layout/DesignToken';
import { COLOR_VARIANTS } from '@lumx/demo/constants';
import { Grid, Theme } from '@lumx/react';

# Colors

**Generic colors to use within your application.**
**Each color has different variants based on opacity.**

## Primary palette

### Blue

Blue color is the primary color for your application.
You can override it by defining another blue color palette in your scss file.
You can also define another color as primary by overrding <code>\$lumx-theme-primary</code> variable.

<DesignTokenGroup>
    {Object.entries(COLOR_VARIANTS['blue']).map(([key, hue], index) => (
        <DesignToken key={index} prefix={<div className={`lumx-theme-background-blue-${key}`} style={{ width: 20, height: 20 }} />} name={`$lumx-theme-blue-${key}`} description={hue.description} version="v0.22">
            <Grid>
                <div className={`lumx-theme-background-blue-${key}`} style={{ width: 60, height: 60 }} />
                <div className="lumx-spacing-margin-left-big lumx-typography-body1 lumx-theme-color-dark-L2">
                    <span className="lumx-base-display-block">{hue.hexCode}</span>
                    <span className="lumx-base-display-block">{hue.rgb}</span>
                    <span className="lumx-base-display-block">{hue.opacity}</span>
                </div>
            </Grid>
        </DesignToken>
    ))}
</DesignTokenGroup>
